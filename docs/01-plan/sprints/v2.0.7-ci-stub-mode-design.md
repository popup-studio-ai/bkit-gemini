### 1.2 모드 분기 로직

```js
const mode = process.env.BKIT_TEST_MODE || 'live';
const hash = computeRequestHash(request);
const fixturePath = path.join(FIXTURE_DIR, `${request.label || 'unlabeled'}-${hash}.json`);

switch (mode) {
  case 'stub':
    if (!fs.existsSync(fixturePath)) {
      throw new Error(
        `[bkit-stub] No fixture for hash=${hash} label=${request.label}.\n` +
        `Run BKIT_TEST_MODE=record to capture, or BKIT_TEST_MODE=live to bypass.`
      );
    }
    return loadFixture(fixturePath).then(f => ({ ...f.response, fromFixture: true }));

  case 'record': {
    const live = await runLive(request);
    saveFixture(fixturePath, { hash, request, response: live, captured_at: new Date().toISOString(), gemini_cli_version: detectVersion() });
    return { ...live, fromFixture: false };
  }

  case 'live':
  default:
    return runLive(request).then(r => ({ ...r, fromFixture: false }));
}
```

### 1.3 `computeRequestHash` — 결정적 직렬화

```js
function computeRequestHash(request) {
  const ALLOWLIST_ENV = [
    'GEMINI_CLI_TRUST_WORKSPACE', 'GEMINI_CLI_VERSION',
    'BKIT_SESSION_START_VERBOSE'
  ];
  const env = {};
  for (const k of ALLOWLIST_ENV.sort()) {
    if (request.env && k in request.env) env[k] = request.env[k];
  }
  const payload = JSON.stringify({
    args: [...request.args],  // 순서 보존 (CLI args는 순서 의미)
    env,                       // 정렬됨
    cwd_rel: request.cwd ? path.basename(request.cwd) : path.basename(process.cwd()),
    bkit_version: getBkitVersion()
  });
  return require('crypto').createHash('sha256').update(payload).digest('hex').slice(0, 16);
}
```

### 1.4 `runLive` — `spawn('gemini', ...)`

stdin 없음, stdout/stderr 캡처, exit code 반환. timeout 적용.

`tests/fixtures/llm-responses/L3-1-onetwothree-<hash>.json` — RCA §1.3 L3-1 시나리오의 합성 응답. 실제 record는 manual 수행 시 자동 update.

## 4. 동적 회귀 (`tests/regression/test-mode/llm-replay.test.js`)

| ID | Description |
|----|-------------|
| LR-01 | computeRequestHash가 동일 입력에 대해 동일 hash 반환 (결정성) |
| LR-02 | args 순서가 다르면 다른 hash (의미 차이 반영) |
| LR-03 | env allowlist 외 key는 hash에 영향 없음 |
| LR-04 | cwd absolute path 변화는 hash에 영향 없음 (basename만 사용) |
| LR-05 | stub mode + fixture 존재 → fixture 응답 반환 (live 호출 0회) |
| LR-06 | stub mode + fixture 부재 → fail-fast with hint |
| LR-07 | live mode → runLive 호출 (mock spawn으로 검증) |
| LR-08 | record mode → fixture 자동 저장 (mock spawn) |
| LR-09 | getTestMode()는 BKIT_TEST_MODE env 정상 반영 |
| LR-10 | listFixtures()는 fixture dir 내 모든 .json 반환 |

stub/record mode test는 `spawn`을 mock해서 LLM 호출 없이 진행.
